# Runtime dependencies for the hosted Streamlit demo.
# PyInstaller, pywebview, pytest, pyarmor — all dev-only — are excluded.
#
# Versions are PINNED for reproducible, supply-chain-safe Cloud rebuilds:
# an unpinned `>=` lets every rebuild silently pull the latest of each package
# (and its transitive deps), so a broken or compromised release lands without
# review. Bumps should arrive as reviewed Dependabot PRs, not surprises.
# Pins match the environment the engine bytecode was compiled and verified
# against (local `streamlit run` + AppTest: all 7 tabs, 0 exceptions). pandas
# is held at 2.3.x — the engine is tested on it, and pandas 3.0 has breaking
# API changes to validate (via a reviewed Dependabot PR) before any bump.
numpy==2.4.4
pandas==2.3.3
matplotlib==3.10.8
scipy==1.17.1
scikit-learn==1.8.0
streamlit==1.55.0
plotly==6.6.0
openpyxl==3.1.5
h5py==3.16.0
# kaleido is intentionally held at the 0.2.x line: it ships a self-contained
# Chromium, so plotly static-image (PNG) export works headless on Streamlit
# Cloud. kaleido 1.x dropped the bundled browser and needs a system Chrome
# that Cloud does not provide — pinning to 1.x would break plot exports.
kaleido==1.3.0
